> stand <- function(x, max, min, option) {
+   
+   if(option == 1) {
+     res <- (x-mean(x, na.rm = TRUE))/sd(x, na.rm = TRUE)
+   }
+   
+   if(option == 2) {
+     res <- (x - min)/(max-min)
+   }
+   
+   return(res)
